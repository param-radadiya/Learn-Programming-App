    void create(struct node **head)
    {
        int c, ch;
        struct node *temp, *rear;
     
        do
        {
            printf("Enter number: ");
            scanf("%d", &c);
            temp = (struct node *)malloc(sizeof(struct node));
            temp->num = c;
            temp->next = NULL;
            if (*head == NULL)
            {
                *head = temp;
            }
            else
            {
                rear->next = temp;
            }
            rear = temp;
            printf("Do you wish to continue [1/0]: ");
            scanf("%d", &ch);
        } while (ch != 0);
        printf("\n");
    }
     
    void display(struct node *p)
    {
        while (p != NULL)
        {
            printf("%d\t", p->num);
            p = p->next;
        }
        printf("\n");
    }
     
    void release(struct node **head)
    {
        struct node *temp = *head;
        *head = (*head)->next;
        while ((*head) != NULL)
        {
            free(temp);
            temp = *head;
            (*head) = (*head)->next;
        }
    }
