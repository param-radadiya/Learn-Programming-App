/*
     * C Program to Check whether a given Number is Perfect Number
    Perfect number is a number which is equal to
    sum of its divisor. For eg,divisors of 6 are
    1,2 and 3. The sum of these divisors is 6. 
    So 6 is called as perfect number.
     */
    #include <stdio.h>
     
    int main()
    {
        int number, rem, sum = 0, i;
     
        printf("Enter a Number\n");
        scanf("%d", &number);
        for (i = 1; i <= (number - 1); i++)
        {
            rem = number % i;
        if (rem == 0)
            {
                sum = sum + i;
            }
        }
        if (sum == number)
            printf("Entered Number is perfect number");
        else
            printf("Entered Number is not a perfect number");
        return 0;
    }
