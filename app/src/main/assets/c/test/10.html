#include <stdio.h>
 
    void main()
    {
        long  num, reverse = 0, temp, remainder;
     
        printf("Enter the number\n");
        scanf("%ld", &num);
        temp = num;
        while (num > 0)
        {
            remainder = num % 10;
            reverse = reverse * 10 + remainder;
            num /= 10;
        }
        printf("Given number = %ld\n", temp);
        printf("Its reverse is = %ld\n", reverse);
    }
    
    >>Case:1
    Enter the number
    567865
    Given number   = 567865
    Its reverse is = 568765
     
    Case:2
    Enter the number
    00001
    Given number   = 00001
    Its reverse is = 10000
