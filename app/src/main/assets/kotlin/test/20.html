fun main(args: Array<String>) {
    val n1 = 72
    val n2 = 120
    var lcm: Int
    // maximum number between n1 and n2 is stored in lcm
    lcm = if (n1 > n2) n1 else n2
    // Always true
    while (true) {
        if (lcm % n1 == 0 && lcm % n2 == 0) {
            println("The LCM of $n1 and $n2 is $lcm.")
            break
        }
        ++lcm
    }
}


>> output will be:

The LCM of 72 and 120 is 360.
