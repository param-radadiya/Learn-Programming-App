import java.util.ArrayList
    fun main(args: Array<String>) {
        val list1 = ArrayList<String>()
        list1.add("a")
        val list2 = ArrayList<String>()
        list2.add("b")
        val joined = ArrayList<String>()
        joined.addAll(list1)
        joined.addAll(list2)
        println("list1: $list1")
        println("list2: $list2")
        println("joined: $joined")
    }



>> output will be:

list1: [a]
list2: [b]
joined: [a, b]
